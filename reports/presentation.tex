\begin{frame}{Project Overview}
    \textbf{Objective:} Develop a highly robust, real-time object detection and temporal state-tracking system for offline inference.
    \vspace{0.4cm}
    
    \textbf{Multi-Layer State Tracking Architecture:}
    \begin{enumerate}
        \item \textbf{Aspect Ratio Gating:} Temporal sustain threshold to prevent noise/flickering.
        \item \textbf{Centroid Drop Analysis:} Identifies state transitions along the optical axis.
        \item \textbf{Projectile-Proximity Gating:} Eliminates environmental false positives; requires the Projectile to be within $K$ pixels of the Target Object within a trailing $M$-frame window.
    \end{enumerate}
\end{frame}

\begin{frame}{Dataset Construction}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{Frame Extraction Protocol}
        \begin{itemize}
            \item Extracted from raw \texttt{.MOV} (30 fps).
            \item \textbf{Stride Strategy:} 1:20 (approx 1.5 fps) to eliminate temporal correlation.
            \item \textbf{Resolution:} Aspect-preserving resize to 1280px via \texttt{cv2.resize}.

\begin{frame}{Critical Data Remediation}
    Initial training yielded catastrophic failure due to malformed Roboflow exports. A custom Python module was engineered to sanitize the dataset:
    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Polygon to BBox Conversion:} Derived extrema ($\min(x), \max(x), \min(y), \max(y)$) to yield standardized YOLO format.
        \item \textbf{Coordinate Clamping:} Strictly clamped values to $[0,1]$ boundary.
        \item \textbf{Extreme Box Filtration:} Removed impossible geometries (w/h $> 0.45$ or $< 0.005$).
    \end{itemize}
    \vspace{0.3cm}
    \textit{Outcome: 3,051 polygons converted, securing a pristine ground truth.}
\end{frame}
